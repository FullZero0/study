#include <iostream>

struct Student {
	string Name;
	int Mat;
	int Fiz;
	int Inf;
	Student(string n=" ", int m=0, int f=0, int i=0) : Name(n), Mat(m), Fiz(f), Inf(i)
	{

	}
	friend std::ostream& operator<< (std::ostream& out, const Student& student);

	int Sredn() {
		return ((Mat + Fiz + Inf) / 3);
	}
};

std::ostream& operator<< (std::ostream& out, const  Student& student)
{
	out << std::setw(9) << std::left << student.Name << std::setw(6) << std::left << student.Mat << std::setw(5) << std::left << student.Fiz << std::setw(5) << std::left << student.Inf;

	return out;
}

	for (auto& value : student)

}

	while (text)

void sortStudent(std::vector<Student>& student)
{
	Student x;
	for (int i = 0; i < (student.size() - 1); i++)
	{
		for (int j = (i+1); j < student.size(); j++)
		{
			if (student[i].Sredn() < student[j].Sredn())
				swap(student[i], student[j]);
		}
	}
}
